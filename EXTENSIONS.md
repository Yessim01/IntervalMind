    """
    –†–∞—Å—á–µ—Ç —Å–ª–µ–¥—É—é—â–µ–≥–æ –∏–Ω—Ç–µ—Ä–≤–∞–ª–∞ –Ω–∞ –æ—Å–Ω–æ–≤–µ –∞–ª–≥–æ—Ä–∏—Ç–º–∞ SM-2

        easiness_factor: —Ñ–∞–∫—Ç–æ—Ä –ª–µ–≥–∫–æ—Å—Ç–∏ (1.3 - 2.5)

    """
    if quality < 3:
        # –ï—Å–ª–∏ –æ—Ç–≤–µ—Ç –ø–ª–æ—Ö–æ–π, –Ω–∞—á–∏–Ω–∞–µ–º —Å–Ω–∞—á–∞–ª–∞

    new_ef = max(1.3, new_ef)  # –ú–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ 1.3
    
    # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º –Ω–æ–≤—ã–π –∏–Ω—Ç–µ—Ä–≤–∞–ª

        return 6
    elif current_interval == 6:
        return int(current_interval * new_ef)
    else:
        return int(current_interval * new_ef)
